## 문자열 내 마음대로 정렬하기
- [Lv.1 문자열 내 마음대로 정렬하기](https://school.programmers.co.kr/learn/courses/30/lessons/12915)
  → [localeCompare()](/blog/javascript/array#localecompare)

```js
function solution(strings, n) {
    var answer = [];
    
    // !callout[/...strings/] 배열 자체를 직접 바꾸는(제자리) 메서드이므로, 얕은 복사(shallow copy)로 원본을 보존
    answer = [...strings].sort((a, b)=> {
        const charA = a[n];
        const charB = b[n];
        
        // 동일하면 문자열 전체 비교
        if (charA === charB) {
            return a.localeCompare(b);
        }
        
        return charA.localeCompare(charB);
    })

    return answer;
}
```

## 폰켓몬
- [Lv.1 폰켓몬](https://school.programmers.co.kr/learn/courses/30/lessons/1845)
  → [Set() → size](/blog/javascript/set#크기-확인)

```js
function solution(nums) {
    var answer = 0;
    
    const max = nums.length / 2
    // !callout[/Set\(nums\)/] 중복 값을 허용하지 않는 점을 이용
    const s = new Set(nums);
    
    answer = max > s.size ? s.size : max
    
    return answer;
}
```
## 완주하지 못한 선수
- [Lv.1 완주하지 못한 선수](https://school.programmers.co.kr/learn/courses/30/lessons/42576) 
  → [Map() → set()](/blog/javascript/hash-map#값-추가-1)

```js
function solution(participant, completion) {
    const map = new Map();
    
    participant.forEach(name => {
        // !callout[/\ \+\ 1/] key로 이름을 설정, 값을 불러온 뒤 없으면 0 → + 1
        map.set(name, (map.get(name) || 0) + 1);
    })
    
    completion.forEach(name => {
        // !callout[/\ \-\ 1/] 목록에 있으면 map에서 값을 불러온 뒤 -1
        map.set(name, (map.get(name) - 1));
    })
    
    for (let [key, val] of map) {
        if (val > 0) return key;
    }
}
```

## 기능 개발
- [Lv.2 기능 개발](https://school.programmers.co.kr/learn/courses/30/lessons/42586)
  → [map()](/blog/javascript/array#mapcallback)
```js
function solution(progresses, speeds) {
  var answer = [];
  
  // 남은 일수를 계산하여 반환
  const days = progresses.map((p, i) => Math.ceil((100 - p) / speeds[i])); 
  let base = days[0];
  let count = 1;
  
  for (let i = 1; i < days.length; i++) {
    // 기준일보다 빨리 끝난 기능 → 추후에 같이 배포
    if (days[i] <= base) {
      count++;
    } else {
      // 기준일보다 늦게 끝남 → 이전 묶음 마감
      answer.push(count);
      base = days[i]; // 새로운 기준일 시작
      count = 1;
    }
  }
  answer.push(count);

  return answer;
}
```
